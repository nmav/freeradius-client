109 int rc_aaa_ctx_server(rc_handle * rh, RC_AAA_CTX ** ctx, SERVER * aaaserver,
110  rc_type type,
111  uint32_t client_port,
112  VALUE_PAIR * send, VALUE_PAIR ** received,
113  char *msg, int add_nas_port,
114  rc_standard_codes request_type)
115 {
116  SEND_DATA data;
117  VALUE_PAIR *adt_vp = NULL;
118  int result;
119  int timeout = rc_conf_int(rh, "radius_timeout");
120  int retries = rc_conf_int(rh, "radius_retries");
121  double start_time = 0;
122  double now = 0;
123  time_t dtime;
124  int servernum;
125 
126  data.send_pairs = send;
127  data.receive_pairs = NULL;
128 
129  /*
130  * if there is more than zero servers, then divide waiting time
131  * among all the servers.
132  */
133  if (aaaserver->max > 0) {
134  if (timeout > 0) {
135  timeout = (timeout + 1) / aaaserver->max;
136  }
137  if (retries > 0) {
138  retries = (retries + 1) / aaaserver->max;
139  }
140  }
141 
142  if (add_nas_port != 0
143  && rc_avpair_get(data.send_pairs, PW_NAS_PORT, 0) == NULL) {
144  /*
145  * Fill in NAS-Port
146  */
147  if (rc_avpair_add(rh, &(data.send_pairs), PW_NAS_PORT,
148  &client_port, 0, 0) == NULL)
149  return ERROR_RC;
150  }
151 
152  if (request_type == PW_ACCOUNTING_REQUEST) {
153  /*
154  * Fill in Acct-Delay-Time
155  */
156  dtime = 0;
157  now = rc_getctime();
158  adt_vp = rc_avpair_get(data.send_pairs, PW_ACCT_DELAY_TIME, 0);
159  if (adt_vp == NULL) {
160  adt_vp = rc_avpair_add(rh, &(data.send_pairs),
161  PW_ACCT_DELAY_TIME, &dtime, 0,
162  0);
163  if (adt_vp == NULL)
164  return ERROR_RC;
165  start_time = now;
166  } else {
167  start_time = now - adt_vp->lvalue;
168  }
169  }
170 
171  if (data.receive_pairs != NULL) {
172  rc_avpair_free(data.receive_pairs);
173  data.receive_pairs = NULL;
174  }
175 
176  servernum = 0;
177  do {
178  rc_buildreq(rh, &data, request_type, aaaserver->name[servernum],
179  aaaserver->port[servernum],
180  aaaserver->secret[servernum], timeout, retries);
181 
182  if (request_type == PW_ACCOUNTING_REQUEST) {
183  dtime = rc_getctime() - start_time;
184  rc_avpair_assign(adt_vp, &dtime, 0);
185  }
186 
187  result = rc_send_server_ctx(rh, ctx, &data, msg, type);
188 
189  if (result == OK_RC) {
190  if (request_type != PW_ACCOUNTING_REQUEST) {
191  *received = data.receive_pairs;
192  } else {
193  rc_avpair_free(data.receive_pairs);
194  }
195 
196  DEBUG(LOG_INFO,
197  "servernum %u returned success", servernum);
198  return result;
199  }
200 
201  rc_avpair_free(data.receive_pairs);
202  data.receive_pairs = NULL;
203 
204  if (radcli_debug) {
205  rc_log(LOG_INFO, "servernum %u returned error: %d (remain: %d)", servernum, result, aaaserver->max-servernum);
206  }
207  servernum++;
208  } while (servernum < aaaserver->max && result == TIMEOUT_RC);
209 
210  return result;
211 }

309 int rc_check(rc_handle * rh, char *host, char *secret, unsigned short port,
310  char *msg)
311 {
312  SEND_DATA data;
313  int result;
314  uint32_t service_type;
315  int timeout = rc_conf_int(rh, "radius_timeout");
316  int retries = rc_conf_int(rh, "radius_retries");
317  rc_type type;
318 
319  data.send_pairs = data.receive_pairs = NULL;
320 
321  if (rh->so_type == RC_SOCKET_TLS || rh->so_type == RC_SOCKET_DTLS)
322  type = AUTH;
323  else
324  type = ACCT;
325 
326  /*
327  * Fill in Service-Type
328  */
329 
330  service_type = PW_ADMINISTRATIVE;
331  rc_avpair_add(rh, &(data.send_pairs), PW_SERVICE_TYPE, &service_type, 0,
332  0);
333 
334  rc_buildreq(rh, &data, PW_STATUS_SERVER, host, port, secret, timeout,
335  retries);
336  result = rc_send_server(rh, &data, msg, type);
337 
338  rc_avpair_free(data.receive_pairs);
339 
340  return result;
341 }
