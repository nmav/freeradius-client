85 static ssize_t tls_sendto(void *ptr, int sockfd,
86  const void *buf, size_t len,
87  int flags, const struct sockaddr *dest_addr,
88  socklen_t addrlen)
89 {
90  tls_st *st = ptr;
91  int ret;
92 
93  if (st->ctx.need_restart != 0) {
94  restart_session(st->rh, st);
95  }
96 
97  ret = gnutls_record_send(st->ctx.session, buf, len);
98  if (ret == GNUTLS_E_AGAIN || ret == GNUTLS_E_INTERRUPTED) {
99  errno = EINTR;
100  return -1;
101  }
102 
103  if (ret < 0) {
104  rc_log(LOG_ERR, "%s: error in sending: %s", __func__,
105  gnutls_strerror(ret));
106  errno = EIO;
107  st->ctx.need_restart = 1;
108  return -1;
109  }
110 
111  st->ctx.last_msg = time(0);
112  return ret;
113 }

129 static ssize_t tls_recvfrom(void *ptr, int sockfd,
130  void *buf, size_t len,
131  int flags, struct sockaddr *src_addr,
132  socklen_t * addrlen)
133 {
134  tls_st *st = ptr;
135  int ret;
136 
137  ret = gnutls_record_recv(st->ctx.session, buf, len);
138  if (ret == GNUTLS_E_AGAIN || ret == GNUTLS_E_INTERRUPTED ||
139  ret == GNUTLS_E_HEARTBEAT_PING_RECEIVED || ret == GNUTLS_E_HEARTBEAT_PONG_RECEIVED) {
140  errno = EINTR;
141  return -1;
142  }
143 
144  if (ret == GNUTLS_E_WARNING_ALERT_RECEIVED) {
145  rc_log(LOG_ERR, "%s: received alert: %s", __func__,
146  gnutls_alert_get_name(gnutls_alert_get(st->ctx.session)));
147  errno = EINTR;
148  return -1;
149  }
150 
151  /* RFC6614 says: "After the TLS session is established, RADIUS packet payloads are
152  * exchanged over the encrypted TLS tunnel. In RADIUS/UDP, the
153  * packet size can be determined by evaluating the size of the
154  * datagram that arrived. Due to the stream nature of TCP and TLS,
155  * this does not hold true for RADIUS/TLS packet exchange.",
156  *
157  * That is correct in principle but it fails to associate the length with
158  * the TLS record boundaries. Here, when in TLS, we assume that a single TLS
159  * record holds a single radius packet. It wouldn't make sense anyway to send
160  * multiple TLS records for a single packet.
161  */
162 
163  if (ret <= 0) {
164  rc_log(LOG_ERR, "%s: error in receiving: %s", __func__,
165  gnutls_strerror(ret));
166  errno = EIO;
167  st->ctx.need_restart = 1;
168  return -1;
169  }
170 
171  st->ctx.last_msg = time(0);
172  return ret;
173 }

178 static int cert_verify_callback(gnutls_session_t session)
179 {
180  unsigned int status;
181  int ret;
182  struct tls_int_st *ctx;
183  gnutls_datum_t out;
184 
185  /* read hostname */
186  ctx = gnutls_session_get_ptr(session);
187  if (ctx == NULL)
188  return GNUTLS_E_CERTIFICATE_ERROR;
189 
190  if (ctx->skip_hostname_check)
191  ret = gnutls_certificate_verify_peers2(session, &status);
192  else
193  ret = gnutls_certificate_verify_peers3(session, ctx->hostname, &status);
194  if (ret < 0) {
195  rc_log(LOG_ERR, "%s: error in certificate verification: %s",
196  __func__, gnutls_strerror(ret));
197  return GNUTLS_E_CERTIFICATE_ERROR;
198  }
199 
200  if (status != 0) {
201  ret =
202  gnutls_certificate_verification_status_print(status,
203  gnutls_certificate_type_get
204  (session),
205  &out, 0);
206  if (ret < 0) {
207  return GNUTLS_E_CERTIFICATE_ERROR;
208  }
209  rc_log(LOG_INFO, "%s: certificate: %s", __func__, out.data);
210  gnutls_free(out.data);
211  return GNUTLS_E_CERTIFICATE_ERROR;
212  }
213 
214  return 0;
215 }

229 static int init_session(rc_handle *rh, tls_int_st *ses,
230  const char *hostname, unsigned port,
231  struct sockaddr_storage *our_sockaddr,
232  int timeout,
233  unsigned secflags)
234 {
235  int sockfd, ret, e;
236  struct addrinfo *info;
237  char *p;
238  unsigned flags = 0;
239  unsigned cred_set = 0;
240  tls_st *st = rh->so.ptr;
241 
242  ses->sockfd = -1;
243  ses->init = 1;
244 
245  pthread_mutex_init(&ses->lock, NULL);
246  sockfd = socket(our_sockaddr->ss_family, (secflags&SEC_FLAG_DTLS)?SOCK_DGRAM:SOCK_STREAM, 0);
247  if (sockfd < 0) {
248  rc_log(LOG_ERR,
249  "%s: cannot open socket", __func__);
250  ret = -1;
251  goto cleanup;
252  }
253 
254  if (our_sockaddr->ss_family == AF_INET)
255  ((struct sockaddr_in *)our_sockaddr)->sin_port = 0;
256  else
257  ((struct sockaddr_in6 *)our_sockaddr)->sin6_port = 0;
258 
259  ses->sockfd = sockfd;
260 
261  /* Initialize DTLS */
262 
263  flags = GNUTLS_CLIENT;
264  if (secflags&SEC_FLAG_DTLS)
265  flags |= GNUTLS_DATAGRAM;
266  ret = gnutls_init(&ses->session, flags);
267  if (ret < 0) {
268  rc_log(LOG_ERR,
269  "%s: error in gnutls_init(): %s", __func__, gnutls_strerror(ret));
270  ret = -1;
271  goto cleanup;
272  }
273 
274  memcpy(&ses->our_sockaddr, our_sockaddr, sizeof(*our_sockaddr));
275  if (!(secflags&SEC_FLAG_DTLS)) {
276  if (timeout > 0) {
277  gnutls_handshake_set_timeout(ses->session, timeout*1000);
278  } else {
279  gnutls_handshake_set_timeout(ses->session, GNUTLS_DEFAULT_HANDSHAKE_TIMEOUT);
280  }
281  } else { /* DTLS */
282  if (timeout > 0)
283  gnutls_dtls_set_timeouts(ses->session, 1000, timeout*1000);
284  }
285 
286  gnutls_transport_set_int(ses->session, sockfd);
287  gnutls_session_set_ptr(ses->session, ses);
288  /* we only initiate heartbeat messages */
289  gnutls_heartbeat_enable(ses->session, GNUTLS_HB_LOCAL_ALLOWED_TO_SEND);
290 
291  p = rc_conf_str(rh, "tls-verify-hostname");
292  if (p && (strcasecmp(p, "false") == 0 || strcasecmp(p, "no"))) {
293  ses->skip_hostname_check = 1;
294  }
295 
296  if (st && st->psk_cred) {
297  cred_set = 1;
298  gnutls_credentials_set(ses->session,
299  GNUTLS_CRD_PSK, st->psk_cred);
300 
301  ret = gnutls_priority_set_direct(ses->session, "NORMAL:-KX-ALL:+ECDHE-PSK:+DHE-PSK:+PSK:-VERS-TLS1.0", NULL);
302  if (ret < 0) {
303  ret = -1;
304  rc_log(LOG_ERR,
305  "%s: error in setting PSK priorities: %s",
306  __func__, gnutls_strerror(ret));
307  goto cleanup;
308 
309  cred_set = 1;
310  }
311  } else if (st) {
312  cred_set = 1;
313  if (st->x509_cred) {
314  gnutls_credentials_set(ses->session,
315  GNUTLS_CRD_CERTIFICATE,
316  st->x509_cred);
317  }
318 
319  gnutls_set_default_priority(ses->session);
320  }
321 
322  gnutls_server_name_set(ses->session, GNUTLS_NAME_DNS,
323  hostname, strlen(hostname));
324 
325  info =
326  rc_getaddrinfo(hostname, PW_AI_AUTH);
327  if (info == NULL) {
328  ret = -1;
329  rc_log(LOG_ERR, "%s: cannot resolve %s", __func__,
330  hostname);
331  goto cleanup;
332  }
333 
334  if (port != 0) {
335  if (info->ai_addr->sa_family == AF_INET)
336  ((struct sockaddr_in *)info->ai_addr)->sin_port =
337  htons(port);
338  else
339  ((struct sockaddr_in6 *)info->ai_addr)->sin6_port =
340  htons(port);
341  } else {
342  rc_log(LOG_ERR, "%s: no port specified for server %s",
343  __func__, hostname);
344  ret = -1;
345  goto cleanup;
346  }
347 
348  strlcpy(ses->hostname, hostname, sizeof(ses->hostname));
349  ses->port = port;
350 
351  if (cred_set == 0) {
352  rc_log(LOG_CRIT,
353  "%s: neither tls-ca-file or a PSK key are configured",
354  __func__);
355  ret = -1;
356  goto cleanup;
357  }
358 
359  /* we connect since we are talking to a single server */
360  ret = connect(sockfd, info->ai_addr, info->ai_addrlen);
361  freeaddrinfo(info);
362  if (ret == -1) {
363  e = errno;
364  ret = -1;
365  rc_log(LOG_CRIT, "%s: cannot connect to %s: %s",
366  __func__, hostname, strerror(e));
367  goto cleanup;
368  }
369 
370  rc_log(LOG_DEBUG,
371  "%s: performing TLS/DTLS handshake with [%s]:%d",
372  __func__, hostname, port);
373  do {
374  ret = gnutls_handshake(ses->session);
375  if (ret == GNUTLS_E_LARGE_PACKET)
376  break;
377  } while (ret < 0 && gnutls_error_is_fatal(ret) == 0);
378 
379  if (ret < 0) {
380  rc_log(LOG_ERR, "%s: error in handshake: %s",
381  __func__, gnutls_strerror(ret));
382  ret = -1;
383  goto cleanup;
384  }
385 
386  return 0;
387  cleanup:
388  deinit_session(ses);
389  return ret;
390 
391 }

397 static void restart_session(rc_handle *rh, tls_st *st)
398 {
399  struct tls_int_st tmps;
400  time_t now = time(0);
401  int ret;
402  int timeout;
403 
404  if (now - st->ctx.last_restart < TIME_ALIVE)
405  return;
406 
407  st->ctx.last_restart = now;
408 
409  timeout = rc_conf_int(rh, "radius_timeout");
410 
411  /* reinitialize this session */
412  ret = init_session(rh, &tmps, st->ctx.hostname, st->ctx.port, &st->ctx.our_sockaddr, timeout, st->flags);
413  if (ret < 0) {
414  rc_log(LOG_ERR, "%s: error in re-initializing DTLS", __func__);
415  return;
416  }
417 
418  if (tmps.sockfd == st->ctx.sockfd)
419  st->ctx.sockfd = -1;
420  deinit_session(&st->ctx);
421  memcpy(&st->ctx, &tmps, sizeof(tmps));
422  st->ctx.need_restart = 0;
423 
424  return;
425 }

466 int rc_check_tls(rc_handle * rh)
467 {
468  tls_st *st;
469  time_t now = time(0);
470  int ret;
471 
472  if (rh->so_type != RC_SOCKET_TLS && rh->so_type != RC_SOCKET_DTLS)
473  return 0;
474 
475  st = rh->so.ptr;
476 
477  if (st->ctx.init != 0) {
478  if (st->ctx.need_restart != 0) {
479  restart_session(rh, st);
480  } else if (now - st->ctx.last_msg > TIME_ALIVE) {
481  ret = gnutls_heartbeat_ping(st->ctx.session, 64, 4, GNUTLS_HEARTBEAT_WAIT);
482  if (ret < 0) {
483  restart_session(rh, st);
484  }
485  st->ctx.last_msg = now;
486  }
487  }
488  return 0;
489 }

511 /*- Initialize a configuration for TLS or DTLS
512  *
513  * This function will initialize the handle for TLS or DTLS.
514  *
515  * @param rh a handle to parsed configuration
516  * @param flags must be zero or SEC_FLAG_DTLS
517  * @return 0 on success, -1 on failure.
518  -*/
519 int rc_init_tls(rc_handle * rh, unsigned flags)
520 {
521  int ret;
522  tls_st *st = NULL;
523  struct sockaddr_storage our_sockaddr;
524  const char *ca_file = rc_conf_str(rh, "tls-ca-file");
525  const char *cert_file = rc_conf_str(rh, "tls-cert-file");
526  const char *key_file = rc_conf_str(rh, "tls-key-file");
527  const char *pskkey = NULL;
528  SERVER *authservers;
529  char hostname[256]; /* server's hostname */
530  unsigned port; /* server's port */
531 
532  memset(&rh->so, 0, sizeof(rh->so));
533 
534  if (flags & SEC_FLAG_DTLS) {
535  rh->so_type = RC_SOCKET_DTLS;
536  rh->so.static_secret = DEFAULT_DTLS_SECRET;
537  } else {
538  rh->so_type = RC_SOCKET_TLS;
539  rh->so.static_secret = DEFAULT_TLS_SECRET;
540  }
541 
542  rc_own_bind_addr(rh, &our_sockaddr);
543 
544  st = calloc(1, sizeof(tls_st));
545  if (st == NULL) {
546  ret = -1;
547  goto cleanup;
548  }
549 
550  st->rh = rh;
551  st->flags = flags;
552 
553  rh->so.ptr = st;
554 
555  if (ca_file || (key_file && cert_file)) {
556  ret = gnutls_certificate_allocate_credentials(&st->x509_cred);
557  if (ret < 0) {
558  ret = -1;
559  rc_log(LOG_ERR,
560  "%s: error in setting X.509 credentials: %s",
561  __func__, gnutls_strerror(ret));
562  goto cleanup;
563  }
564 
565  if (ca_file) {
566  ret =
567  gnutls_certificate_set_x509_trust_file(st->x509_cred,
568  ca_file,
569  GNUTLS_X509_FMT_PEM);
570  if (ret < 0) {
571  ret = -1;
572  rc_log(LOG_ERR,
573  "%s: error in setting X.509 trust file: %s",
574  __func__, gnutls_strerror(ret));
575  goto cleanup;
576  }
577  }
578 
579  if (cert_file && key_file) {
580  ret =
581  gnutls_certificate_set_x509_key_file(st->x509_cred,
582  cert_file,
583  key_file,
584  GNUTLS_X509_FMT_PEM);
585  if (ret < 0) {
586  ret = -1;
587  rc_log(LOG_ERR,
588  "%s: error in setting X.509 cert and key file: %s",
589  __func__, gnutls_strerror(ret));
590  goto cleanup;
591  }
592  }
593 
594  gnutls_certificate_set_verify_function(st->x509_cred,
595  cert_verify_callback);
596  }
597 
598  /* Read the PSK key if any */
599  authservers = rc_conf_srv(rh, "authserver");
600  if (authservers == NULL) {
601  rc_log(LOG_ERR,
602  "%s: cannot find authserver", __func__);
603  ret = -1;
604  goto cleanup;
605  }
606  if (authservers->max > 1) {
607  ret = -1;
608  rc_log(LOG_ERR,
609  "%s: too many auth servers for TLS/DTLS; only one is allowed",
610  __func__);
611  goto cleanup;
612  }
613  strlcpy(hostname, authservers->name[0], sizeof(hostname));
614  port = authservers->port[0];
615  if (authservers->secret)
616  pskkey = authservers->secret[0];
617 
618  if (pskkey && pskkey[0] != 0) {
619  char *p;
620  char username[64];
621  gnutls_datum_t hexkey;
622  int username_len;
623 
624  if (strncmp(pskkey, "psk@", 4) != 0) {
625  ret = -1;
626  rc_log(LOG_ERR,
627  "%s: server secret is set but does not start with 'psk@'",
628  __func__);
629  goto cleanup;
630  }
631  pskkey+=4;
632 
633  if ((p = strchr(pskkey, '@')) == NULL) {
634  ret = -1;
635  rc_log(LOG_ERR,
636  "%s: PSK key is not in 'username@hexkey' format",
637  __func__);
638  goto cleanup;
639  }
640 
641  username_len = p - pskkey;
642  if (username_len + 1 > sizeof(username)) {
643  rc_log(LOG_ERR,
644  "%s: PSK username too big", __func__);
645  ret = -1;
646  goto cleanup;
647  }
648 
649  strlcpy(username, pskkey, username_len + 1);
650 
651  p++;
652  hexkey.data = (uint8_t*)p;
653  hexkey.size = strlen(p);
654 
655  ret = gnutls_psk_allocate_client_credentials(&st->psk_cred);
656  if (ret < 0) {
657  ret = -1;
658  rc_log(LOG_ERR,
659  "%s: error in setting PSK credentials: %s",
660  __func__, gnutls_strerror(ret));
661  goto cleanup;
662  }
663 
664  ret =
665  gnutls_psk_set_client_credentials(st->psk_cred,
666  username, &hexkey,
667  GNUTLS_PSK_KEY_HEX);
668  if (ret < 0) {
669  ret = -1;
670  rc_log(LOG_ERR,
671  "%s: error in setting PSK key: %s",
672  __func__, gnutls_strerror(ret));
673  goto cleanup;
674  }
675  }
676 
677  ret = init_session(rh, &st->ctx, hostname, port, &our_sockaddr, 0, flags);
678  if (ret < 0) {
679  ret = -1;
680  goto cleanup;
681  }
682 
683  rh->so.get_fd = tls_get_fd;
684  rh->so.sendto = tls_sendto;
685  rh->so.recvfrom = tls_recvfrom;
686  rh->so.lock = tls_lock;
687  rh->so.unlock = tls_unlock;
688  return 0;
689  cleanup:
690  if (st) {
691  if (st->ctx.init != 0)
692  deinit_session(&st->ctx);
693  if (st->x509_cred)
694  gnutls_certificate_free_credentials(st->x509_cred);
695  if (st->psk_cred)
696  gnutls_psk_free_client_credentials(st->psk_cred);
697  }
698  free(st);
699  return ret;
700 }
