22 /*- Returns a struct addrinfo from a host name or address in textual notation.
23  *
24  * @param host the name of the host
25  * @param flags should be a combinations of PW_AI flags
26  * @return address which should be deallocated using freeaddrinfo() or NULL on failure
27  -*/
28 struct addrinfo *rc_getaddrinfo (char const *host, unsigned flags)
29 {
30  struct addrinfo hints, *res;
31  int err;
32  const char *service = NULL;
33 
34  memset(&hints, 0, sizeof(hints));
35  hints.ai_socktype = SOCK_DGRAM;
36  if (flags & PW_AI_PASSIVE)
37  hints.ai_flags = AI_PASSIVE;
38 
39  if (flags & PW_AI_AUTH)
40  service = "radius";
41  else if (flags & PW_AI_ACCT)
42  service = "radacct";
43 
44  err = getaddrinfo(host, service, &hints, &res);
45  if (err != 0) {
46  return NULL;
47  }
48 
49  return res;
50 }

122 int rc_get_srcaddr(struct sockaddr *lia, const struct sockaddr *ria)
123 {
124  int temp_sock;
125  socklen_t namelen;
126 
127  temp_sock = socket(ria->sa_family, SOCK_DGRAM, 0);
128  if (temp_sock == -1) {
129  rc_log(LOG_ERR, "rc_get_srcaddr: socket: %s", strerror(errno));
130  return -1;
131  }
132 
133  if (connect(temp_sock, ria, SA_LEN(ria)) != 0) {
134  rc_log(LOG_ERR, "rc_get_srcaddr: connect: %s",
135  strerror(errno));
136  close(temp_sock);
137  return -1;
138  }
139 
140  namelen = SA_LEN(ria);
141  if (getsockname(temp_sock, lia, &namelen) != 0) {
142  rc_log(LOG_ERR, "rc_get_srcaddr: getsockname: %s",
143  strerror(errno));
144  close(temp_sock);
145  return -1;
146  }
147 
148  close(temp_sock);
149  return 0;
150 }
